DROP POLICY IF EXISTS "storage_allow_all_authenticated" ON storage.objects;

CREATE POLICY "storage_tenant_scoped"
ON storage.objects
FOR ALL
TO authenticated
USING (
  bucket_id = 'work-order-photos'
  AND (storage.foldername(name))[1] = (
    SELECT tenant_id::text FROM public.users WHERE id = auth.uid()
  )
)
WITH CHECK (
  bucket_id = 'work-order-photos'
  AND (storage.foldername(name))[1] = (
    SELECT tenant_id::text FROM public.users WHERE id = auth.uid()
  )
);
